# TalentFlow-AI Cloud Firestore Data Model & Schema

## 1. Top-Level Collections Overview

Cloud Firestore is structured into top-level collections for primary domain entities, with subcollections for candidate pipeline stage evaluation artifacts and interview logs.

```
Cloud Firestore
├── users/{userId}
├── recruiters/{recruiterId}
├── jobs/{jobId}
├── candidates/{candidateId}
│   ├── resume_analysis/{analysisId}
│   ├── matching/{matchId}
│   ├── background/{bgId}
│   └── decision/{decisionId}
├── interviews/{interviewId}
│   ├── proctoring_logs/{logId}
│   ├── speech/{speechId}
│   └── technical/{techId}
├── reports/{reportId}
├── notifications/{notificationId}
└── activity_logs/{logId}
```

## 2. Collection Schemas & Field Definitions

### 2.1 Collection: `users`
Stores system user profiles and role assignments.

- **Document ID**: `{userId}` (Matches Firebase Auth `uid`)
- **Fields**:
```typescript
interface UserDocument {
  id: string;                 // Firebase Auth UID
  name: string;               // Full name
  email: string;              // User email address
  role: "admin" | "recruiter" | "candidate";
  company?: string;           // Company name (for recruiters)
  created_at: Timestamp;      // User creation time
  updated_at: Timestamp;      // Last update timestamp
}
```

### 2.2 Collection: `recruiters`
Stores recruiter-specific profiles and active job associations.

- **Document ID**: `{recruiterId}`
- **Fields**:
```typescript
interface RecruiterDocument {
  id: string;                 // Recruiter unique ID
  name: string;               // Recruiter name
  email: string;              // Recruiter email
  company: string;            // Company name
  role: string;               // Job title/role (e.g., "Talent Lead")
  active_jobs: string[];      // Array of job IDs managed by this recruiter
  created_at: Timestamp;      // ISO/Timestamp
}
```

### 2.3 Collection: `jobs`
Stores job postings and criteria for AI matching.

- **Document ID**: `{jobId}`
- **Fields**:
```typescript
interface JobRequirements {
  skills: string[];           // Required technical/soft skills
  experience_years: number;   // Minimum required experience in years
  education?: string;         // Minimum education degree
  location?: string;          // Job location or "Remote"
}

interface JobDocument {
  id: string;                 // Job unique ID
  title: string;              // Job title
  description: string;        // Full job description
  requirements: JobRequirements;
  recruiter_id: string;       // Foreign key -> recruiters.id
  status: "open" | "closed" | "draft";
  created_at: Timestamp;      // Creation timestamp
  application_count: number;  // Number of candidates applied
  pipeline_stages: string[];  // Active pipeline stage workflow definitions
}
```

### 2.4 Collection: `candidates`
Stores candidate profile details and application status.

- **Document ID**: `{candidateId}`
- **Fields**:
```typescript
interface CandidateDocument {
  id: string;                 // Candidate unique ID
  name: string;               // Candidate full name
  email: string;              // Candidate email address
  phone?: string;             // Contact phone number
  resume_url?: string;        // Firebase Storage URL of uploaded resume PDF
  pipeline_stage: "applied" | "screening" | "recruiter_review" | "interview_scheduled" | "interview_completed" | "decision" | "hired" | "rejected";
  status: "active" | "withdrawn" | "rejected" | "hired" | "hold";
  applied_job_id: string;     // Foreign key -> jobs.id
  created_at: Timestamp;      // Application submission time
  updated_at: Timestamp;      // Last pipeline status update time
  metadata: Record<string, any>; // Arbitrary metadata (source, UTM params)
}
```

#### Subcollection: `candidates/{candidateId}/resume_analysis/{analysisId}`
Stores parsed resume structure and ATS scoring generated by **Resume Agent**.

```typescript
interface ResumeAnalysisDocument {
  id: string;
  parsed_text: string;
  skills: string[];
  projects: Array<{ title: string; description: string; tech_stack: string[] }>;
  education: Array<{ degree: string; institution: string; year: number }>;
  companies: string[];
  experience: Array<{ company: string; role: string; duration: string }>;
  certifications: string[];
  ats_score: number;          // 0.0 - 100.0
  resume_score: number;       // 0.0 - 100.0
  missing_keywords: string[];
  analyzed_at: Timestamp;
}
```

#### Subcollection: `candidates/{candidateId}/matching/{matchId}`
Stores semantic fit analysis generated by **Matching Agent**.

```typescript
interface MatchingResultDocument {
  id: string;
  job_id: string;
  overall_match_score: number; // 0.0 - 1.0 (or 0 - 100%)
  skill_match_score: number;
  experience_match_score: number;
  missing_skills: string[];
  matching_highlights: string[];
  gap_analysis: string;
  created_at: Timestamp;
}
```

#### Subcollection: `candidates/{candidateId}/background/{bgId}`
Stores employment/education background check artifacts generated by **Background Agent**.

```typescript
interface BackgroundCheckDocument {
  id: string;
  verification_status: "verified" | "flagged" | "pending";
  employment_verified: boolean;
  education_verified: boolean;
  flags: string[];
  verification_summary: string;
  checked_at: Timestamp;
}
```

#### Subcollection: `candidates/{candidateId}/decision/{decisionId}`
Stores synthesized hiring decision produced by **Decision Agent**.

```typescript
interface HiringDecisionDocument {

  job_id: string;
  recommendation: "strong_hire" | "hire" | "maybe" | "no_hire" | "strong_no_hire";
```

### 2.5 Collection: `interviews`
Stores scheduled and completed interview sessions.

- **Document ID**: `{interviewId}`
- **Fields**:
```typescript
interface InterviewTranscriptEntry {
  speaker: "ai" | "candidate";
  text: string;
  timestamp: string;
}

interface InterviewDocument {
  id: string;                 // Interview ID
  candidate_id: string;       // Foreign key -> candidates.id
  job_id: string;             // Foreign key -> jobs.id
  scheduled_at: Timestamp;    // Scheduled start time
  duration_minutes: number;   // Scheduled duration (default 60)
  status: "scheduled" | "in_progress" | "completed" | "canceled" | "no_show";
  transcript: InterviewTranscriptEntry[];
  proctoring_flags: string[]; // ["multiple_faces", "looking_away", etc.]
  interviewer_type: "ai" | "human";
  created_at: Timestamp;
}
```

#### Subcollection: `interviews/{interviewId}/proctoring_logs/{logId}`
Stores real-time OpenCV proctoring anomaly events.

```typescript
interface ProctoringLogDocument {
  id: string;
  timestamp: Timestamp;
  flag: "multiple_faces" | "no_face" | "looking_away" | "background_noise" | "tab_switch";
  confidence: number;
  snapshot_url?: string;
}
```

### 2.6 Collection: `reports`
Stores comprehensive candidate evaluation reports.

- **Document ID**: `{reportId}`
- **Fields**:
```typescript
interface ReportDocument {
  id: string;
  candidate_id: string;       // Foreign key -> candidates.id
  job_id: string;             // Foreign key -> jobs.id
  scores: Record<string, number>; // e.g. {"resume": 88.5, "matching": 92.0, "technical": 85.0}
  recommendation: "strong_hire" | "hire" | "maybe" | "no_hire" | "strong_no_hire";
  remarks: string;
  generated_at: Timestamp;
  pdf_url?: string;
}
```

### 2.7 Collection: `activity_logs`
System activity and candidate stage transition logs.

- **Document ID**: `{logId}`
- **Fields**:
```typescript
interface ActivityLogDocument {
  id: string;
```

## 3. Firestore Security Rules (`firestore.rules`)

```javascript
rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {

    // Helper functions
    function isAuthenticated() {
      return request.auth != null;
    }
    
    function getUserData() {
      return get(/databases/$(database)/documents/users/$(request.auth.uid)).data;
    }
    
    function isAdmin() {
      return isAuthenticated() && getUserData().role == 'admin';
    }
    
    function isRecruiter() {
      return isAuthenticated() && (getUserData().role == 'recruiter' || getUserData().role == 'admin');
    }

    // Users Collection
    match /users/{userId} {
      allow read: if isAuthenticated();
      allow write: if isAdmin() || request.auth.uid == userId;
    }

    // Recruiters Collection
    match /recruiters/{recruiterId} {
      allow read: if isAuthenticated();
      allow write: if isRecruiter();
    }

    // Jobs Collection
    match /jobs/{jobId} {
      allow read: if true; // Public job listings
      allow write: if isRecruiter();
    }

    // Candidates Collection & Subcollections
    match /candidates/{candidateId} {
      allow read: if isRecruiter() || (isAuthenticated() && request.auth.uid == candidateId);
      allow create: if true; // Public application submission
      allow update, delete: if isRecruiter();

      match /{subcollection=**} {
        allow read, write: if isRecruiter();
      }
    }

    // Interviews Collection & Subcollections
    match /interviews/{interviewId} {
      allow read: if isRecruiter() || isAuthenticated();
      allow write: if isRecruiter();

      match /{subcollection=**} {
        allow read, write: if isRecruiter();
      }
    }

    // Reports Collection
    match /reports/{reportId} {
      allow read, write: if isRecruiter();
    }

    // Activity Logs Collection
    match /activity_logs/{logId} {
      allow read: if isRecruiter();
      allow write: if isAuthenticated();
    }
  }
}
```
